# Frontend Dependencies Installation Guide

# First, navigate to the project directory
cd c:\Users\Admin\Desktop\111

# Install core Next.js and React dependencies
npm install next react react-dom

# Install charting library and its TypeScript types
npm install --save lightweight-charts
npm install --save @types/lightweight-charts

# Install HTTP client for API calls
npm install --save axios

# Install TypeScript and type definitions
npm install --save-dev typescript @types/react @types/node

# Note: After installation, you may need to run:
# npm run dev - to start the development server
# npm run build - to create a production build
